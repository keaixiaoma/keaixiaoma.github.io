幸运数

暴力去枚举区间里面的数肯定是不行的。这道题要用一个小的数学方法：容斥
$$
在[1,x]区间内能被a整除的数字的个数为x/a, 记c为a和b的最小公倍数，
$$

$$
利用容斥原理计算a,b,c在区间内出现了多少次，用a和b出现次数之和减掉c出现次数之和即可
$$

#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll l, r;
ll calc(ll x) {
    return r / x - (l - 1) / x;
}
ll gcd(ll a, ll b) {
    return b ? gcd(b, a % b) : a;
}
int main() {
    int t;
    scanf("%d", &t);
    while (t--){
        int a, b;
        scanf("%d%d%lld%lld", &a, &b, &l, &r);
        ll LCM = (ll)a * b / gcd(b, a % b);
        printf("%lld\n", calc(a) + calc(b) - calc(LCM));
    }
    return 0;
}

算法社の年终奖品（easy version）

这道题对于n的数据范围比较小，所以可以直接暴力枚举每一个宿舍楼作为放年终奖品的地方，先用floyd预处理每一个宿舍之间的距离即可，是一道几乎裸的多源最短路。

#include <bits/stdc++.h>
#define inf 0x7fffffff
using namespace std;
const int maxn = 1e2+10;

int mp[maxn][maxn];
int dis[maxn][maxn];
int w[maxn];
int main(){
    int n;
    cin >> n;
    for(int i = 1; i <= n; ++i){
        for(int j = i; j <= n; ++j){
            if(i == j) mp[i][j] = 0;
            else mp[i][j] = inf, mp[j][i] = inf;
        }
    }
    for(int i = 1; i <= n; ++i){
        int a, b, c;
        cin >> a >> b >> c;
        w[i] = a;
        if(b) mp[i][b] = 1, mp[b][i] = 1;
        if(c) mp[i][c] = 1, mp[c][i] = 1;
    }
    for(int i = 1; i <= n; ++i){
        for(int j = 1; j <= n; ++j){
            for(int k = 1; k <= n; ++k){
                if(mp[j][i] != inf && mp[i][k] != inf)
                     mp[j][k] = min(mp[j][k], mp[j][i]+mp[i][k]);
            }
        }
    }
    int ans = inf;
    for(int i = 1; i <= n; ++i){
        int ff = 0;
        for(int j = 1; j <= n; ++j){
            ff += w[j] * mp[i][j];
        }
        ans = min(ans, ff);
    }
    cout << ans;
    return 0;
}

还有一种BFS O(n2)算法 对每个点求解

出题人没写这种算法代码，就用某个人的考场代码了

#include<bits/stdc++.h>
using namespace std;
const int maxn=105,inf=1e8+5;
int n,w,u,v,mi,data[maxn],mp[maxn][maxn],e[maxn][3],vis[maxn];
queue<int> q;
void bfs(int x){
    q.push(x);
    q.push(0);
    int sum=0;
    while(!q.empty()){
        int a=q.front();
        q.pop();
        int s=q.front();
        q.pop();
        vis[a]=1;
        sum+=s*data[a];
        for(int i=0;i<3;++i){
            if(!vis[e[a][i]]&&e[a][i]){
            q.push(e[a][i]);
            q.push(s+1);
            }
        }
    }
    if(sum<mi) mi=sum;
}
int main(){
    cin>>n;
    mi=inf;
    for(int i=1;i<=n;++i){
        cin>>w>>u>>v;
        data[i]=w;
        e[i][0]=u;
        e[i][1]=v;
        if(u){
            mp[u][i]=1;
            mp[i][u]=1;
            e[u][2]=i;
        }
        if(v){
            mp[v][i]=1;
            mp[i][v]=1;
            e[v][2]=i;
        }
    }
    for(int i=1;i<=n;++i){
        for(int j=1;j<=n;++j) vis[j]=0;
        bfs(i);
    }
    cout<<mi;
}

幸运变量

由题可知，每个变量建立关系之后就是一个整体，不需要纠结内部的关系，所以这是一道明摆的并查集，维护每个变量集合，用一个同时记录了大小和属性的并查集维护即可。

#include <bits/stdc++.h>
using namespace std;
#define N 100001
int f[N], num[N], col[N];
int n, m, ans;
int find(int x) {
    return f[x] == x ? x : f[x] = find(f[x]);
}
void merge(int x, int y) {
    int fx = find(x), fy = find(y);
    if (fx == fy) {
        if (!col[fx]) ans += num[fx];
        col[fx] = 1;
        return;
    }
    if (!col[fx]) swap(fx, fy);
    if (col[fx] && !col[fy]) ans += num[fy];
    f[fy] = fx;
    num[fx] += num[fy];
    num[fy] = 0;
}
int main() {
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i++) f[i] = i, num[i] = 1, col[i] = 0;
    for (int i = 1, x, y; i <= m; i++) {
        scanf("%d%d", &x, &y);
        merge(x, y);
        printf("%d\n", ans);
    }
    return 0;
}

还有二叉树的记忆化搜索也能过qwq不过还算优秀的解法

这道题的正解为：带权树的重心。

树的重心的定义：

树若以某点为根，使得该树最大子树的结点数最小，那么这个点则为该树的重心，一棵树可能有多个重心。

树的重心的性质：

1、树上所有的点到树的重心的距离之和是最短的，如果有多个重心，那么总距离相等。

2、插入或删除一个点，树的重心的位置最多移动一个单位。

3、若添加一条边连接2棵树，那么新树的重心一定在原来两棵树的重心的路径上。

当然，这题我们只需要用到第一条性质。

树的重心的解法这里就不多加赘述了，如果感兴趣的话可以自行去了解（毕竟也是个数据结构的性质），用的算法是树状dp，是个优雅的O（n）复杂度的算法，下面给出AC代码

#include <bits/stdc++.h>
#define ll long long
using namespace std;
const int maxn = 1e5+10;
struct node{
    int v, next;
}tree[maxn];

int head[maxn], cnt, w[maxn];
int siz[maxn];
int dp[maxn];
int ans = 0x7fffffff;
void add(int from, int to){
    tree[++cnt].next = head[from];
    tree[cnt].v = to;
    head[from] = cnt;
}

void dfs(int u, int fa, int depth){
    siz[u] += w[u];
    for(int i = head[u]; i; i = tree[i].next){
        int v = tree[i].v;
        if(v != fa){
            dfs(v, u, depth+1);
            siz[u] += siz[v];
        }
    }
    dp[1] += w[u] * depth;
}

void f(int u, int fa){
    for(int i = head[u]; i; i = tree[i].next){
        int v = tree[i].v;
        if(v != fa){
            dp[v] = dp[u] - siz[v] + siz[1] - siz[v];
            f(v, u);
        }
    }
    ans = min(ans, dp[u]);
}
int main() {
    int n;
    cin >> n;
    for(int i = 1; i <= n; ++i){
        int a, b, c;
        cin >> a >> b >> c;
        w[i] = a;
        if(b) add(i, b), add(b, i);
        if(c) add(i, c), add(c, i);
    }
    dfs(1, 0, 0);
    f(1, 0);
    cout << ans;
    return 0;
}

高中の排列组合问题

如题面所说，确实是一道高中的排列组合（想必大家高中都做过）

假设我们考虑到第n个信封，初始时我们暂时把第n封信放在第n个信封中，然后考虑两种情况的递推：

  • 前面n - 1个信封全部装错；
  • 前面n - 1个信封有一个没有装错其余全部装错。

对于第一种情况，前面n - 1个信封全部装错：因为前面n - 1个已经全部装错了，所以第n封只需要与前面任一一个位置交换即可，总共有 f(n-1)*(n-1)种情况。

对于第二种情况，前面n - 1个信封有一个没有装错其余全部装错：考虑这种情况的目的在于，若n - 1个信封中如果有一个没装错，那么我们把那个没装错的与n交换，即可得到一个全错位排列情况。

其他情况，我们不可能通过一次操作来把它变成一个长度为n的错排。

于是可得错位排列的递推式为f(n) = (n - 1) * (f(n - 1) + f(n - 2))。

#include <bits/stdc++.h>

#define ll long long
#define inf 0x3f3f3f
using namespace std;
const int maxn = 1e5+10;

ll f[maxn];
ll mod = 1000000007;
int main(){
    f[2] = 1;
    for(int i = 3; i <= maxn; ++i){
        f[i] = ((i-1)*((f[i-1]+f[i-2])%mod))%mod;
    }
    int t;
    cin >> t;
    while(t--){
        int n;
        cin >> n;
        cout << f[n] << endl;
    }
    return 0;
}
